### Analysis Scripts

While I perform the analysis internally, here are the scripts you can use to run the same validation locally on a repository.

Save the following two files in the *same* directory, then run the `run_scan.sh` script from your terminal, pointing it to the repo you want to scan.

**Usage:**

```bash
chmod +x run_scan.sh
./run_scan.sh /path/to/your/gpt-api-intel-main
```

#### 1\. `run_scan.sh`

This is a helper script to run the Python scanner.

```bash
#!/bin/bash
# Script to scan a repository for file references

# Check if Python 3 is installed
if ! command -v python3 &> /dev/null
then
    echo "Python 3 could not be found. Please install Python 3."
    exit 1
fi

# Check if a directory is provided
if [ -z "$1" ]
then
    echo "Usage: $0 <path_to_repo_directory>"
    echo "Example: $0 ./gpt-api-intel-main"
    exit 1
fi

REPO_PATH=$1

echo "Running scan on: $REPO_PATH"

# Run the python script
python3 scan_repo_links.py "$REPO_PATH"

echo "Scan complete. Results saved to:"
echo "- updated_paths.json"
echo "- basenames_not_found.json"
echo "- all_references_found.json"
```

#### 2\. `scan_repo_links.py`

This is the Python script that performs the complete scan and validation.

```python
import os
import json
import re
import argparse
from collections import defaultdict

# Regex to find potential file paths and filenames.
# This will match strings like:
# 'foo.json', 'data/bar.py', '../src/baz.md', 'a.b.schema.json'
# It avoids matching http/https URLs to reduce false positives.
FILE_REGEX = re.compile(
    r"""
    (?<!https?://)          # Negative lookbehind: not preceded by 'http://' or 'https://'
    (                       # Start capture group 1
      (?:[\w\-\.]+\/)+      # Match one or more directory levels (e.g., 'data/', '../src/', 'a.b/')
      |                     # OR
      (?<![\w\-\.\/])       # Negative lookbehind: ensure it's not part of a longer path
    )
    (                       # Start capture group 2: the base filename
      [\w\-\.]+             # Match the filename part (e.g., 'foo', 'bar.config')
      \.                    # Match the literal dot
      [a-zA-Z0-9_]{2,}      # Match the extension (at least 2 chars)
    )
    (?![/\w])               # Negative lookahead: not followed by more path chars
    """,
    re.VERBOSE
)

# Directories to ignore during the scan
IGNORE_DIRS = {'.git', 'node_modules', '__pycache__', '.vscode'}
# File extensions to ignore (e.g., binaries)
IGNORE_EXTENSIONS = {'.png', '.jpg', '.jpeg', '.gif', '.woff', '.woff2', '.eot', '.ttf', '.svg'}

def get_all_files(repo_path):
    """Walks the repo directory and returns a set of all normalized file paths."""
    file_list = set()
    base_path = os.path.normpath(repo_path)
    
    for root, dirs, files in os.walk(base_path, topdown=True):
        # Filter out ignored directories
        dirs[:] = [d for d in dirs if d not in IGNORE_DIRS]
        
        for file in files:
            file_path = os.path.join(root, file)
            # Filter out ignored file types
            if os.path.splitext(file)[1].lower() not in IGNORE_EXTENSIONS:
                file_list.add(os.path.normpath(file_path))
                
    return file_list

def analyze_repo(repo_path):
    """Scans all files in the repo, validates references, and generates JSON reports."""
    
    # Normalize the base repo path
    repo_path = os.path.normpath(repo_path)
    
    # 1. Get all files in the repository
    print(f"Scanning for all files in {repo_path}...")
    all_files = get_all_files(repo_path)
    if not all_files:
        print(f"Error: No files found in directory: {repo_path}")
        return

    # 2. Create a set of all basenames for quick lookup
    all_basenames = {os.path.basename(f) for f in all_files}
    print(f"Found {len(all_files)} files and {len(all_basenames)} unique basenames.")

    # 3. Initialize result lists
    updated_paths = []
    not_found = []
    all_references_found = []
    
    # Keep track of unique finds to avoid duplicates
    seen_not_found = set()
    seen_updated = set()
    seen_found = set()

    print("Analyzing file contents for references...")
    # 4. Iterate through each file and scan its content
    for file_path in all_files:
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                for line_num, line in enumerate(f, 1):
                    
                    # Find all matches in the current line
                    for match in FILE_REGEX.finditer(line):
                        mentioned_path = (match.group(1) or "") + match.group(2)
                        base_filename = match.group(2)

                        # Skip self-references
                        if base_filename == os.path.basename(file_path):
                            continue

                        # Create a unique key for this reference
                        ref_key = (file_path, line_num, mentioned_path)

                        # --- Validation Logic ---
                        
                        # Case 1: Base filename is not in the repo at all
                        if base_filename not in all_basenames:
                            if ref_key not in seen_not_found:
                                not_found.append({
                                    "file_it_was_found_in": os.path.relpath(file_path, repo_path),
                                    "base_filename": base_filename,
                                    "line_number": line_num,
                                    "path": mentioned_path
                                })
                                seen_not_found.add(ref_key)
                            continue

                        # Base filename *was* found. Now check the path.
                        current_file_dir = os.path.dirname(file_path)
                        resolved_path = os.path.normpath(os.path.join(current_file_dir, mentioned_path))
                        
                        # Case 2: The exact, resolved path exists
                        if resolved_path in all_files:
                            if ref_key not in seen_found:
                                all_references_found.append({
                                    "file_it_was_found_in": os.path.relpath(file_path, repo_path),
                                    "base_filename": base_filename,
                                    "line_number": line_num,
                                    "path": mentioned_path
                                })
                                seen_found.add(ref_key)
                        
                        # Case 3: Path is wrong, but basename exists elsewhere
                        else:
                            if ref_key not in seen_updated:
                                # Find the first actual path for this basename
                                try:
                                    actual_path = next(f for f in all_files if os.path.basename(f) == base_filename)
                                    actual_path_rel = os.path.relpath(actual_path, repo_path)
                                except StopIteration:
                                    # This shouldn't happen if all_basenames is correct, but as a safeguard
                                    continue 

                                updated_paths.append({
                                    "file_it_was_found_in": os.path.relpath(file_path, repo_path),
                                    "base_filename": base_filename,
                                    "line_number": line_num,
                                    "old_path": mentioned_path,
                                    "new_path": actual_path_rel
                                })
                                seen_updated.add(ref_key)

        except (IOError, UnicodeDecodeError) as e:
            print(f"Warning: Could not read file {file_path}. Skipping. Error: {e}")

    # 5. Write the JSON output files
    print("Writing output files...")
    with open('updated_paths.json', 'w', encoding='utf-8') as f:
        json.dump(updated_paths, f, indent=2)
        
    with open('basenames_not_found.json', 'w', encoding='utf-8') as f:
        json.dump(not_found, f, indent=2)
        
    with open('all_references_found.json', 'w', encoding='utf-8') as f:
        json.dump(all_references_found, f, indent=2)

def main():
    parser = argparse.ArgumentParser(description="Scan a repository for file references and validate them.")
    parser.add_argument("repo_path", type=str, help="The root directory of the repository to scan.")
    args = parser.parse_args()
    
    if not os.path.isdir(args.repo_path):
        print(f"Error: Path provided is not a valid directory: {args.repo_path}")
        return
        
    analyze_repo(args.repo_path)

if __name__ == "__main__":
    main()
```
